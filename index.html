Extra: Feeling time improperly is a common problem for people with ADHD. There is a bunch of ways to fix it and set this timer is one of them.

function updateTimer() {
 var dt = Date.now() - expected
  let seconds = time % 60
  let minutes = Math.floor(time / 60)
  seconds = seconds < 10 ? '0' + seconds : seconds
  minutes = minutes < 10 ? '0' + minutes : minutes
  countdownElement.innerHTML = `${minutes}:${seconds}`
  if (time >= finalTime && !mainScreenOn){
   hourPassed()
  else {
   time++
   expected += interval;
   setTimeout(updateTimer, Math.max(0, interval - dt))
  }
}
